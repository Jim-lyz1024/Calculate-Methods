 float *ColPivot(float*c,int n)
 {
     int i,j,t,k;
     float *x,p;
     x=(float *)malloc(n*sizeof(float));

       for(j=0;j<n;j++)
         {
             for(t=0;t<=n;t++)
             {
                 printf("%7.2f", *(c+j*(MAX_V+1)+t));
             }
             printf("\n");
         }
      printf("\n");
     for(i=0;i<=n-2;i++)
     {
         k=i;
         for(j=i+1;j<n-1;j++)
         {
             if(fabs(*(c+j*(MAX_V+1)+i))>fabs(*(c+k*(MAX_V+1)+i)))
              k=j;
         }
         if(k!=i)
            for(j=i;j<=n;j++)
         {
             p=*(c+i*(MAX_V+1)+j);
             *(c+i*(MAX_V+1)+j)=*(c+k*(MAX_V+1)+j);
             *(c+k*(MAX_V+1)+j)=p;
         }

         for(j=i+1;j<=n-1;j++)
         {
             p=(*(c+j*(MAX_V+1)+i))/(*(c+i*(MAX_V+1)+i));
             for(t=i;t<=n;t++)
             {
                 *(c+j*(MAX_V+1)+t)-=p*(*(c+i*(MAX_V+1)+t));
             }
         }

         for(j=i;j<n;j++)
         {
             for(t=i;t<=n;t++)
             {
                 printf("%7.2f", *(c+j*(MAX_V+1)+t));
             }
             printf("\n");
         }
         printf("\n");

     }
     for(i=n-1;i>=0;i--)
     {
         for(j=n-1;j>=i+1;j--)
         {
             (*(c+i*(MAX_V+1)+n))-=x[j]*(*(c+i*(MAX_V+1)+j));
         }
         x[i]=*(c+i*(MAX_V+1)+n)/(*(c+i*(MAX_V+1)+i));
     }
     return x;
 }
